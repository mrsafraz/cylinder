Basic Dialog

Show Basic Dialog 

<a href="#" click.delegate="showBasicDialog()">Show Basic Dialog</a>


showBasicDialog(){
	Dialog.showInstance('cylinder/demo/dialogs/my-greeting', {name: 'Cylinder'}, {
		autoclose: true,
		size: 'large', // 'small', default 'medium'
	});
}

Popover

Show Popover 

<a href="#" click.delegate="showPopover($event)">Show Popover</a>


showPopover($event){
	Dialog.showInstance('cylinder/demo/dialogs/my-greeting', {name: 'Cylinder'}, {
		popover: $event.target,
		autoclose: true,
	});
}

Action Sheet

Show Action Sheet 

<a href="#" click.delegate="showActionSheet($event)">Show Action Sheet</a>


showActionSheet($event){
	Dialog.showActionSheet(['Comment', 'Reply', 'Delete', 'Cancel'], 'Actions Title', {
		popover: $event.target,
		autoclose: true,
	}).then((action)=> {
		if(action == 'Reply'){
			alert('Replying...');
		}
		if(action == 'Comment'){
			alert('Commenting...');
		}
	});
}
